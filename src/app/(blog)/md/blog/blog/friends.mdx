```sh
POST http://localhost:3000/api/friends
Authorization

{
  "title": "Found a bug", 
  "name": "Friend 1", # required

  "avatar": "https://friend1.com/avatar.png",
  "desc": "This is friend 1's website.",
  "labels": [
    "lost" 
  ]
}
```
